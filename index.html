        // Add an option

        function addOption(option) {

            const optionWithoutEmojis = trimmedOption.replace(/[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{1F1E0}-\u{1F1FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}\u{FE0F}]/gu, '').trim();


            // Check if the option (without emojis) already exists in mealOptions

            const exists = mealOptions.some(existingOption =>

            );


            if (trimmedOption && !exists) {

            }

        }

                const exists = mealOptions.some(existingOption =>

                    existingOption.replace(/[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{1F1E0}-\u{1F1FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}\u{FE0F}]/gu, '').trim() === mealWithoutEmojis

                );

                    addedCount++;

        // Function to search Google Maps

        function searchGoogleMaps(mealName) {

            // Remove emojis from mealName for the search query

            const mealNameForSearch = mealName.replace(/[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{1F1E0}-\u{1F1FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}\u{FE0F}]/gu, '').trim();

            // Encode only the meal name

            const query = encodeURIComponent(mealNameForSearch);

            // Construct the standard Google Maps URL

            const googleMapsUrl = `https://www.google.com/maps/search/${query}`;

            console.log("Opening Google Maps URL:", googleMapsUrl); // Debugging line

            window.open(googleMapsUrl, '_blank'); // Open in new tab

        }


        // Function to get complementary item from Gemini API

        async function getComplementaryItem(mealName) {

            const geminiDrinkBtn = document.getElementById('getGeminiDrinkBtn');

            const originalButtonText = geminiDrinkBtn.textContent;

            geminiDrinkBtn.textContent = '載入中...';

            geminiDrinkBtn.disabled = true; // Disable button during loading


            // Remove emojis from mealName for the prompt to avoid confusing the LLM

            const mealNameForPrompt = mealName.replace(/[\u{1F600}-\u{1F64F}\u{1F300}-\u{1F5FF}\u{1F680}-\u{1F6FF}\u{1F1E0}-\u{1F1FF}\u{2600}-\u{26FF}\u{2700}-\u{27BF}\u{FE0F}]/gu, '').trim();


            try {

                let chatHistory = [];

                // Modified prompt to request a concise answer within 200 characters and without bullet points/stars

                const prompt = `請為我建議一個與 "${mealNameForPrompt}" 搭配的飲品或配菜。請用繁體中文回答，限制在200字以內，不需要使用星號或列表格式。`;

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""; // Canvas will automatically provide the API key at runtime

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });


                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    let text = result.candidates[0].content.parts[0].text;

                    // Further clean up any remaining asterisks or excessive newlines

                    text = text.replace(/\*+/g, '').replace(/\n\s*\n/g, '\n').trim(); // Remove multiple asterisks and extra newlines

                    showMessageBox(text); // Display Gemini's response in the message box

                } else {

                    showMessageBox('未能獲取搭配建議，請稍後再試。');

                }

            } catch (error) {

                console.error('Error calling Gemini API for complementary item:', error);

                showMessageBox('獲取搭配建議時發生錯誤。');

            } finally {

                geminiDrinkBtn.textContent = originalButtonText; // Restore button text

                geminiDrinkBtn.disabled = false; // Re-enable button

            }

        }


        // Event listeners

        addOptionBtn.addEventListener('click', () => {

            addOption(newOptionInput.value);

        });


        newOptionInput.addEventListener('keypress', (e) => {

            if (e.key === 'Enter') {

                addOption(newOptionInput.value);

            }

        });


        optionsList.addEventListener('click', (e) => {

            if (e.target.classList.contains('remove-btn')) {

                const indexToRemove = parseInt(e.target.dataset.index);

                removeOption(indexToRemove);

            }

        });


        predefinedOptionsDiv.addEventListener('click', (e) => {

            if (e.target.classList.contains('add-predefined-btn')) {

                addOption(e.target.textContent);

            }

        });


        // Event listener for the "Add All Options" button

        addAllOptionsBtn.addEventListener('click', addAllPredefinedOptions);

        // Event listener for the new "Remove All Options" button

        removeAllOptionsBtn.addEventListener('click', removeAllOptions);


        pickMealBtn.addEventListener('click', pickRandomMeal);


        closeMessageBoxBtn.addEventListener('click', hideMessageBox);
